# Folder Structure

```
apps/                         # This repository (clawde-io/apps)
│
├── daemon/                   # clawd — Rust/Tokio daemon
│   ├── src/
│   │   ├── main.rs           # Entry point — starts the WebSocket server
│   │   ├── lib.rs            # Crate root
│   │   ├── config/           # Config loading (TOML + env)
│   │   ├── ipc/              # JSON-RPC 2.0 dispatcher and handlers
│   │   ├── session/          # Session lifecycle + AI runner trait
│   │   ├── repo/             # Git2 integration, filesystem watcher
│   │   └── db/               # SQLite schema + sqlx queries
│   └── Cargo.toml
│
├── desktop/                  # Flutter desktop app (macOS / Windows / Linux)
│   ├── lib/
│   │   ├── main.dart         # Entry point
│   │   ├── app.dart          # Root widget + router
│   │   └── features/
│   │       ├── layout/       # Multi-pane shell, sidebar
│   │       ├── sessions/     # Session list (uses clawd_ui SessionListTile)
│   │       ├── chat/         # Chat view (uses clawd_ui ChatBubble + MessageInput)
│   │       ├── editor/       # CodeMirror 6 WebView integration
│   │       └── settings/     # App settings
│   ├── macos/                # macOS runner + entitlements
│   ├── windows/              # Windows runner
│   ├── linux/                # Linux runner
│   └── pubspec.yaml
│
├── mobile/                   # Flutter mobile app (iOS + Android)
│   ├── lib/
│   │   ├── main.dart         # Entry point
│   │   ├── app.dart          # Root widget + router + bottom nav shell
│   │   └── features/
│   │       ├── sessions/     # Session list screen
│   │       ├── session_detail/ # Chat + tool-call approval screen
│   │       └── settings/     # Settings screen
│   ├── ios/                  # iOS runner (generated by flutter create)
│   ├── android/              # Android runner (generated by flutter create)
│   └── pubspec.yaml
│
├── packages/
│   ├── clawd_proto/          # Dart: protocol types
│   │   └── lib/src/
│   │       ├── session.dart  # Session, SessionStatus, ProviderType
│   │       ├── message.dart  # Message, MessageRole
│   │       ├── tool_call.dart # ToolCall, ToolResult, ToolCallStatus
│   │       └── rpc.dart      # RpcRequest, RpcResponse, RpcError, ClawdError
│   │
│   ├── clawd_client/         # Dart: WebSocket/JSON-RPC client
│   │   └── lib/src/
│   │       ├── client.dart   # ClawdClient — connect, call<T>, pushEvents stream
│   │       └── exceptions.dart # ClawdRpcError, ClawdDisconnectedError
│   │
│   ├── clawd_core/           # Dart: shared Riverpod state
│   │   └── lib/src/providers/
│   │       ├── daemon_provider.dart   # DaemonNotifier, daemonProvider
│   │       ├── session_provider.dart  # SessionListNotifier, activeSessionIdProvider
│   │       ├── message_provider.dart  # MessageListNotifier (family)
│   │       └── tool_call_provider.dart # ToolCallNotifier (family)
│   │
│   └── clawd_ui/             # Dart: shared Flutter widgets
│       └── lib/src/
│           ├── theme/clawd_theme.dart          # Brand colours + ThemeData
│           └── widgets/
│               ├── chat_bubble.dart            # ChatBubble
│               ├── session_list_tile.dart      # SessionListTile
│               ├── tool_call_card.dart         # ToolCallCard (with approve/reject)
│               ├── message_input.dart          # MessageInput bar
│               ├── connection_status_indicator.dart
│               └── provider_badge.dart         # ProviderBadge (Claude / Codex / …)
│
├── .github/
│   ├── workflows/ci.yml      # Dart analyze+test · cargo clippy+test
│   └── ISSUE_TEMPLATE/       # Bug report + feature request forms
│
├── .wiki/                    # GitHub Wiki source (synced on push)
│
├── melos.yaml                # Dart monorepo workspace manager
├── .gitignore
├── README.md
└── LICENSE
```
